#include<iostream>
#include<stdexcept>
int main() {
	double a;
	double b;

	std::cout << "Unesite prvi broj: ";
	std::cin >> a;
	std::cout << "Unesite drugi broj: ";
	std::cin >> b;

	try {
		if (b == 0) {
			throw b;
		}
		else {
			
			std::cout << (a / b);
		}
	}
	catch (double) {
		std::cout << "Druga vrijednsot ne smije biti nula";
	}
}
